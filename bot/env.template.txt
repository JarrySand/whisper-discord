# Discord Bot Configuration
DISCORD_BOT_TOKEN=your_discord_bot_token_here
DISCORD_CLIENT_ID=your_discord_client_id_here

# =============================================================================
# Audio Processing
# =============================================================================
# 無音検知閾値（ms）- 発話の区切りを検出する時間
# 短いと頻繁に分断、長いと文脈が繋がる（推奨: 1000-2000ms）
AUDIO_SILENCE_THRESHOLD=1500
# 最大セグメント長（ms）- これより長い発話は分割される
# 長いほどAPI呼び出し回数が減り、コスト効率・文脈反映が良い
# ※ Whisper推奨上限は30秒。低メモリ環境（256MB以下）では 5000 以下を推奨
AUDIO_MAX_SEGMENT_DURATION=30000
# 最小セグメント長（ms）- これより短い発話は破棄される
AUDIO_MIN_SEGMENT_DURATION=1000
# 最小RMS閾値 - これ以下の音量のセグメントは破棄（API コスト削減）
# 0.005 = 非常に静か、0.01 = 静か、0.02 = 通常会話
AUDIO_MIN_RMS_THRESHOLD=0.005

# =============================================================================

# =============================================================================
OUTPUT_LOG_DIR=./logs

OUTPUT_ENABLE_DISCORD_POST=true
OUTPUT_ENABLE_FILE_LOG=true
OUTPUT_ENABLE_JSON_STORE=true
OUTPUT_ENABLE_MARKDOWN=true

# Discord Output Format
OUTPUT_DISCORD_FORMAT=standard
OUTPUT_DISCORD_SHOW_TIMESTAMP=true
OUTPUT_DISCORD_SHOW_CONFIDENCE=false
OUTPUT_DISCORD_BATCH_MESSAGES=true
OUTPUT_DISCORD_BATCH_INTERVAL_MS=3000

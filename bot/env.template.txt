# Discord Bot Configuration
DISCORD_BOT_TOKEN=your_discord_bot_token_here
DISCORD_CLIENT_ID=your_discord_client_id_here

# =============================================================================
# Whisper API Configuration
# =============================================================================
# Bot は Whisper API サーバーに接続して文字起こしを行います。
# プロバイダーの切り替えは Whisper API サーバー側 (whisper-api/.env) で設定します。
#
# 【プロバイダーの種類】
#   local   - ローカル faster-whisper (large-v3, 無料, CPU/GPU)
#   groq    - Groq API (large-v3, 超高速, $0.0001/秒)
#   openai  - OpenAI API (whisper-1, 安定, $0.006/分)
#
# 詳細は whisper-api/env.example を参照してください。
# =============================================================================

# Whisper API サーバーの URL
# ローカルサーバー: http://localhost:8000
# 外部サーバー: https://your-whisper-server.example.com
WHISPER_API_URL=http://localhost:8000
WHISPER_API_TIMEOUT=60000

# Audio Processing
# 無音検知閾値（ms）- 発話の区切りを検出する時間
AUDIO_SILENCE_THRESHOLD=800
# 最大セグメント長（ms）- これより長い発話は分割される
AUDIO_MAX_SEGMENT_DURATION=8000
# 最小セグメント長（ms）- これより短い発話は破棄される
AUDIO_MIN_SEGMENT_DURATION=1000
# 最小RMS閾値 - これ以下の音量のセグメントは破棄（API コスト削減）
# 0.005 = 非常に静か、0.01 = 静か、0.02 = 通常会話
AUDIO_MIN_RMS_THRESHOLD=0.005

# Output Configuration
OUTPUT_LOG_DIR=./logs
OUTPUT_SEGMENT_DIR=./segments
OUTPUT_ENABLE_DISCORD_POST=true
OUTPUT_ENABLE_FILE_LOG=true
OUTPUT_ENABLE_JSON_STORE=true
OUTPUT_ENABLE_MARKDOWN=true

# Discord Output Format
OUTPUT_DISCORD_FORMAT=standard
OUTPUT_DISCORD_SHOW_TIMESTAMP=true
OUTPUT_DISCORD_SHOW_CONFIDENCE=false
OUTPUT_DISCORD_BATCH_MESSAGES=true
OUTPUT_DISCORD_BATCH_INTERVAL_MS=3000

# SQLite Search Feature (optional)
# サーバーごとに個別のDBファイルが作成されます（./data/guild_{guildId}.db）
ENABLE_SQLITE=false
SQLITE_DB_DIR=./data
SQLITE_CLEANUP_DAYS=30

# Logging
LOG_LEVEL=info
